b) Mostrar las consultas a las que asistió el paciente, también debe mostrar el medico que lo atendió, en que consultorio, el diagnóstico y la patología detectada. 

GRANT SELECT ON consulta TO perezj_informes;
GRANT SELECT ON empleado TO perezj_informes;
GRANT SELECT ON consultorio TO perezj_informes;
GRANT SELECT ON diagnostico TO perezj_informes;
GRANT SELECT ON patologia TO perezj_informes;


SELECT pa.id_paciente,
       p.nombre AS nombre_paciente,
       p.apellido AS apellido_paciente,
       c.fecha AS fecha_consulta,
       per_med.nombre AS nombre_medico,
       per_med.apellido AS apellido_medico,
       cons.nombre AS consultorio,
       d.descripcion AS diagnostico,
       pat.nombre AS patologia
FROM consulta c
JOIN paciente pa ON c.id_paciente = pa.id_paciente
JOIN persona p ON pa.id_paciente = p.id_persona
JOIN empleado e ON c.id_empleado = e.id_empleado
JOIN persona per_med ON e.id_persona = per_med.id_persona
JOIN consultorio cons ON c.id_consultorio = cons.id_consultorio
JOIN diagnostico d ON c.id_consulta = d.id_consulta
JOIN patologia pat ON d.id_patologia = pat.id_patologia
WHERE p.nombre = 'Luna' AND p.apellido = 'Giralt'
ORDER BY c.fecha;

c) Mostrar las internaciones que tuvo el paciente, debe mostrar la habitación y la cama en la que estuvo. 


GRANT SELECT ON internacion TO perezj_informes;
GRANT SELECT ON habitacion TO perezj_informes;
GRANT SELECT ON cama TO perezj_informes;

SELECT pe.nombre AS nombre_paciente,
       pe.apellido AS apellido_paciente,
       i.fecha_ingreso,
       i.fecha_alta,
       h.numero AS habitacion,
       c.numero AS cama
FROM internacion i
JOIN paciente pa ON i.id_paciente = pa.id_paciente
JOIN persona pe ON pa.id_persona = pe.id_persona
JOIN habitacion h ON i.id_habitacion = h.id_habitacion
JOIN cama c ON i.id_cama = c.id_cama
WHERE pe.nombre = 'Luna' AND pe.apellido = 'Giralt'
ORDER BY i.fecha_ingreso;



d) Mostrar los estudios que le realizaron, los equipos que se utilizaron y el profesional que le realizo el estudio. 

GRANT SELECT ON estudio_realizado TO perezj_informes;
GRANT SELECT ON estudio TO perezj_informes;
GRANT SELECT ON equipo TO perezj_informes;

SELECT pe.nombre AS nombre_paciente,
       pe.apellido AS apellido_paciente,
       er.fecha,
       te.nombre AS tipo_estudio,
       eq.nombre AS equipo,
       per_prof.nombre AS nombre_profesional,
       per_prof.apellido AS apellido_profesional
FROM estudio_realizado er
JOIN paciente pa ON er.id_paciente = pa.id_paciente
JOIN persona pe ON pa.id_persona = pe.id_persona
JOIN tipo_estudio te ON er.id_tipo_estudio = te.id_tipo_estudio
JOIN equipo eq ON er.id_equipo = eq.id_equipo
JOIN empleado em ON er.id_empleado = em.id_empleado
JOIN persona per_prof ON em.id_persona = per_prof.id_persona
WHERE pe.nombre = 'Luna' AND pe.apellido = 'Giralt'
ORDER BY er.fecha;

a) Muestre el nombre, apellido y la obra social de todos los pacientes (incluso de aquellos que no tienen obra social).

FROM paciente pa

b) Liste el nombre, apellido, especialidad y sueldo de todos los empleados que trabajan por la Noche. 

SELECT pe.nombre, pe.apellido, es.nombre AS especialidad, em.sueldo
FROM empleado em
JOIN persona pe ON em.id_persona = pe.id_persona
JOIN especialidad es ON em.id_especialidad = es.id_especialidad
JOIN trabajan tr ON em.id_empleado = tr.id_empleado
WHERE tr.turno = 'Noche';

c) Mostrar los tratamientos cuyo número de dosis sea mayor que 4, es decir, el nombre del 
medicamento, presentación, clasificación y descripción del tratamiento. 

SELECT m.nombre AS medicamento, m.presentacion, cl.nombre AS clasificacion, t.descripcion
FROM tratamiento t
JOIN medicamento m ON t.id_medicamento = m.id_medicamento
JOIN clasificacion cl ON m.id_clasificacion = cl.id_clasificacion
WHERE t.numero_dosis > 4;

d) Mostrar todas las facturas que han sido pagadas parcialmente. 

SELECT f.id_factura, f.fecha, f.monto_total, f.monto_pagado
FROM factura f
WHERE f.monto_pagado > 0 AND f.monto_pagado < f.monto_total;

Ejercicio nro. 3:  
Implemente la seguridad a través de esquemas, para ello: 
a) Cree un nuevo esquema en la base de datos Hospital, con el nombre de “facturacion”. 

CREATE SCHEMA facturacion;

b) Mueva las tablas pago y factura al nuevo esquema “facturacion” (lo puede hacer desde la pestaña general de las propiedades de tabla). 
-- Esto se hace cambiando el esquema de las tablas. En SQL sería:

ALTER TABLE pago SET SCHEMA facturacion;
ALTER TABLE factura SET SCHEMA facturacion;

c) Conéctese al servidor con el usuario de facturación y realice nuevamente la consulta E. Explique si la pudo ejecutar y cómo lo hizo.

-- La consulta E era: “Mostrar todos los pagos realizados por ‘Sara Rosell’”.
Ahora que las tablas están en el esquema facturacion, el usuario perezj_facturacion_user debe referirse a ellas con el prefijo del esquema

SELECT pa.id_pago, pa.fecha, pa.monto, pe.nombre, pe.apellido
FROM facturacion.pago pa
JOIN paciente p ON pa.id_paciente = p.id_paciente
JOIN persona pe ON p.id_persona = pe.id_persona
WHERE pe.nombre = 'Sara' AND pe.apellido = 'Rosell';

-- Explicación
- El usuario perezj_facturacion_user tiene permisos de SELECT sobre las tablas pago y factura.
- Al moverlas al esquema facturacion, debe usarse el nombre calificado (facturacion.pago, facturacion.factura).
- Si el esquema facturacion se agrega al search_path del usuario, podría omitirse el prefijo y seguir usando pago y factura directamente:
ALTER ROLE perezj_facturacion_user SET search_path TO facturacion, public;

ALTER ROLE perezj_facturacion_user SET search_path TO facturacion, public;
